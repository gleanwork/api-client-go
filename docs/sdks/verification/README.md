)

func main() {
    ctx := context.Background()

    s := apiclientgo.New(
        apiclientgo.WithSecurity(os.Getenv("GLEAN_API_TOKEN")),
    )

    res, err := s.Client.Verification.AddReminder(ctx, components.ReminderRequest{
        DocumentID: "<id>",

    if err != nil {
        log.Fatal(err)

}
