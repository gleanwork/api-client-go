## Upload

Upload images for various types (profile pic, background, UGC thumnail/content, etc) with additional metadata.

### Example Usage

```go
package main

import(
	"context"
	"os"
	apiclientgo "github.com/gleanwork/api-client-go"
	"github.com/gleanwork/api-client-go/models/operations"
	"log"
)

func main() {
    ctx := context.Background()

    s := apiclientgo.New(
        apiclientgo.WithSecurity(os.Getenv("GLEAN_BEARER_AUTH")),
    )

    requestBody, fileErr := os.Open("example.file")
    if fileErr != nil {
        panic(fileErr)
    }


    res, err := s.Client.Images.Upload(ctx, operations.UploadimageRequest{
        RequestBody: requestBody,
    })
    if err != nil {
        log.Fatal(err)
    }
    if res.UploadImageResponse != nil {
        // handle response
    }
}
```
